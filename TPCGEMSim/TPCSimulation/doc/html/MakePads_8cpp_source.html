84 void MakePads::Amplify(int evtnr,double xe, double ye ,double te, TPCVoxel* voxel,
85  vector<TPCVoxel> *voxelvector, int particlenr, bool Is_Curler)
86 {
87  isCurler=Is_Curler;
88  CalculateNrElectrons();
89 
90  if(numbersecel>0)
91  {
92  //map position of electron into GEM hole, pitch 140 mum
93  //pitch in y=sqrt(3/4)*pitch because of hexagonal structure
94  double xe_gem=0.;
95  double ye_gem = (int)(ye/(sqrt(3./4.)*GEMPITCH)+(ye<0 ? -0.5:0.5))*sqrt(3./4.)*GEMPITCH;
96  int yint=(int)(ye/(sqrt(3./4.)*GEMPITCH)+0.5);
97  if(yint%2)
98  xe_gem = static_cast<int>(xe/GEMPITCH+(xe<0 ? -0.5:0.5))*GEMPITCH;
99  else
100  //everysecond row is shifted with half a pitch
101  xe_gem = (static_cast<int>(xe/GEMPITCH)+(xe<0 ? -0.5:0.5))*GEMPITCH;
102 
103  if(fabs(xe-xe_gem)>0.08)
104  cout<<"WARNING: x GEM pitch failed, difference: "<<fabs(xe-xe_gem)<<" "<<xe<<endl;
105  if(fabs(ye-ye_gem)>0.08)
106  cout<<"WARNING: y GEM pitch failed, difference: "<<fabs(ye-ye_gem)<<" "<<ye<<endl;
107 
108  double sigma=sigma_diff; // charge broadening between GEMs
109 
110  //start values for integration
111  double x1=xe_gem-4*sigma;
112  double y1=ye_gem-4*sigma;
113  //end values for integration
114  double x2=xe_gem+4*sigma;
115  double y2=ye_gem+4*sigma;
116  // start and end in columns and rows
117  int column1=(int)((x0+x1)/padx);
118  int row1=(int)((y0+y1)/pady);
119  int column2=(int)((x0+x2)/padx);
120  int row2=(int)((y0+y2)/pady);
121 
122  int rownumbers=row2-row1+1;
123  int columnnumbers=column2-column1+1;
124 
125  // time of electron in ns (no additional smearing in time is done here)
126  int time = (int)(te*1000);
127  // integration loop
128  for(int yint=0;yint<rownumbers;yint++)
129  {
130  for(int xint=0;xint<columnnumbers;xint++)
131  {
132  double cstart=(column1+xint)*padx-x0;
133  double cend=(column1+1+xint)*padx-x0;
134  double rend=(row1+1+yint)*pady-y0;
135  double rstart =(row1+yint)*pady-y0;
136  // integration over a 2d gaussian
137  int charge=(int)(numbersecel/4*((erf((cend-xe_gem)/(sqrt(2.)*sigma))-
138  erf((cstart-xe_gem)/(sqrt(2.)*sigma)))*
139  (erf((rend-ye_gem)/(sqrt(2.)*sigma))-
140  erf((rstart-ye_gem)/(sqrt(2.)*sigma)))));
141 
142  long int channr=(row1+yint)*padcolumns+(column1+xint);
143  //check if e- drifted out of read out region
144  if((row1+yint) >=0 && (column1+xint)>=0 &&
145  (column1+xint)<padcolumns && (row1+yint)<(int)(npads/padcolumns) &&
146  charge >0 && channr<npads)
147  {
148  voxel->SetVoxel(channr,time,charge,particlenr);
149  voxelvector->push_back(*voxel);
150  }
151  }
152  }// integration over pads
153  }

163 double MakePads::GEM_Gain(int GEMNr, int enumber)
164 {
165  //Gain
166  double gainmean=Gain[GEMNr-1]/gainfactor; //factor for gain correction to fit data
167  // exponential distribution for gain distribution of single electron
168  double electrons=0.;
169  if(enumber<200)
170  {
171  for(int i=0;i<enumber;i++)
172  electrons+=gsl_ran_exponential(r,gainmean);
173  }
174  else //because than statistics is enough to form gaussian anyway
175  electrons=(gsl_ran_gaussian(r,sqrt((double)enumber))+enumber)*gainmean;
176 
177  if(electrons>=0)
178  return electrons;
179  else
180  return 0;
181 }
