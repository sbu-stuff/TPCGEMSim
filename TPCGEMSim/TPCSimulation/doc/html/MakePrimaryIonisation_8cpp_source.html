67 void MakePrimaryIonisation::CreateTrack(int eventnr, int particlenr, TPCSimTrack *track,
68  TParticle *rootParticle, bool isCurler)
69 {
70  SetParticleParameters(rootParticle);
71  InitializeTrackPoint();
72  InTPC=0;
73  int e_curlercounter=0;
74  CalculateNumberOfClusters();
75  // as long as cluster in TPC and particle not stopped (p>0) continue track
76  while((xend+xv)*(xend+xv)+(yend+yv)*(yend+yv)<=TPCRadius*TPCRadius &&
77  fabs(zend+zv)<fabs(distance) && p>0.0)
78  {
79  if(energy>mass)
80  {
81  CalculateClusterDistance();
82  double delta_cut=0;
83  if(B==0)
84  {
85  CalculateTrackPointwithoutBfield();
86  delta_cut=60; // equals range of 100 mum
87  }
88  if(B!=0)
89  {
90  CalculateTrackPointwithBfield();
91  delta_cut=(30.0*B+0.017)/0.026;
92  //delta_cut=1200 equals 100 mum at B=1T
93  }
94  // Warning if huge rather big steps occur, might result in empty pads
95  if(dr>10 && dp>1) cout<<"Warning: high dE/dx "<<endl;
96  CalculateNumberOfElectrons();
97 
98  //delta electron that moves away from track
99  if(numberofelectrons > delta_cut)
100  {
101  double delta_range=0;
102  double delta_energy=(0.026*numberofelectrons-0.017)*pow(10.,-6.);//in GeV
103  //angle to track in most cases 90 degrees
104  double delta_angle_track=acos(sqrt(delta_energy/energy));// cos of angle
105  double delta_angle_plane=2*M_PI*drand48();
106 
107  if(B==0)
108  delta_range=8.23935e-05*pow(numberofelectrons,1.7293); //mm
109  if(B!=0)
110  delta_range=(delta_energy/(0.3*B))*1000; //in mm
111 
112  double step=delta_range/(double)numberofelectrons;
113 
114  double xe=xend+xv;
115  double ye=yend+yv;
116  double ze=zend+zv;
117 
118  //direction if track would be in 0,1,0
119  double deltax=sin(delta_angle_track)*cos(delta_angle_plane);
120  double deltaz=cos(delta_angle_track);
121  double deltay=sin(delta_angle_track)*sin(delta_angle_plane);
122  //transform to track coordinates
123  RotateToTrack(deltax,deltay,deltaz);
124 
125  for(int enumber=0;enumber<numberofelectrons;enumber++)
126  {
127 
128  //Multiple Scattering
129  if(!(enumber%30) && !(enumber==0))
130  {
131  double scatter_phi=2*M_PI*drand48();
132  double scatter_theta=1./6*M_PI*drand48();
133  // new direction
134  MultipleScattering(scatter_phi,scatter_theta);
135  }
136  if(sqrt((xe+delta_x*step)*(xe+delta_x*step)+(ye+delta_y*step)*(ye+delta_y*step))>=inner_radius)
137  {
138  if(isCurler)
139  {
140  if(e_curlercounter%10==0) //take only every 10th electron for curlers
141  {
142  track->AddTPCElectron(eventnr,particlenr,xe+delta_x*step,ye+delta_y*step,ze+delta_z*step,tend);
143  InTPC++; // counter for # e-
144  }
145  e_curlercounter++;
146  }
147  else
148  {
149  track->AddTPCElectron(eventnr,particlenr,xe+delta_x*step,ye+delta_y*step,ze+delta_z*step,tend);
150  InTPC++;
151  }
152  }
153 
154  xe+=delta_x*step;
155  ye+=delta_y*step;
156  ze+=delta_z*step;
157  }
158  }
159  else
160  {
161  for(int enumber=0;enumber<numberofelectrons;enumber++)
162  {
163  if(sqrt((xend+xv)*(xend+xv)+(yend+yv)*(yend+yv))>=inner_radius)
164  {
165  if(isCurler)
166  {
167  if(e_curlercounter%10==0)//take only every 10th electron for curlers
168  {

171  }
172  e_curlercounter++;
173  }
174  else
175  {
176  track->AddTPCElectron(eventnr,particlenr,xend+xv,yend+yv,zend+zv,tend);
177  InTPC++;
178  }
179  }
180  }
181  }
182  CalculateEnergyLoss();
183  SetNextStartPoint();

185  }
186  else

217 void MakePrimaryIonisation::CalculateTrackPointwithBfield()
218 {
219  // rotation angle
220  double factor = charge*B*C;
221  alpha=factor/p*clusterdistance/1000.; //[1/mm]
222  // center of circle
223  xm=1000*py*dp/factor; //[mm]
224  ym=-1000*px*dp/factor;
225  //coordinates on helix
226  xend=cos(alpha)*(xa-xm)*dp+sin(alpha)*(ya-ym)*dp+xm;
227  yend=-sin(alpha)*(xa-xm)*dp+cos(alpha)*(ya-ym)*dp+ym;
228  zend=za+pz/p*clusterdistance;
229 
230 }

244 void MakePrimaryIonisation::CalculateNumberOfClusters()
245 {
246  //from parametrisation one knows function (polynom) and constants
247  double x=sqrt((energy*energy)/(mass*mass)-1);
248  double logx = log(x);
249  double log10000 = log(10000.0);
250  if(x<10000)
251  numberofclusters = a0+a1*logx+a2*logx*logx+a3*pow(logx,3)
252  +a4*pow(logx,4)+a5*pow(logx,5)
253  +a6*pow(logx,6)+a7*pow(logx,7);
254  else
255  numberofclusters = a0+a1*log10000+a2*log10000*log10000+a3*pow(log10000,3)
256  +a4*pow(log10000,4)+a5*pow(log10000,5)
257  +a6*pow(log10000,6)+a7*pow(log10000,7);
258  // number of cluster per cm constant after p/m>10.000
259 }

324 void MakePrimaryIonisation::RotateToTrack(double x, double y, double z)
325 {
326  double phi;
327  if(px>=0)
328  phi=atan(py/px);
329  else
330  phi=atan(py/px)+M_PI;
331  if(px==0 && py<0)
332  phi=-phi;
333  double theta=acos(pz);
334  delta_x=cos(phi)*cos(theta)*x-sin(phi)*y+cos(phi)*sin(theta)*z;
335  delta_y=sin(phi)*cos(theta)*x+cos(phi)*y+sin(phi)*sin(theta)*z;
336  delta_z=-sin(theta)*x+cos(theta)*z;
337 }
