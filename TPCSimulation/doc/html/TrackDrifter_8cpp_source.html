85 void TrackDrifter::MakeElectronDriftCoordinates()
86 {
87  //drift electron
88  double driftdistance = distance-z_e;
89  double meandrift = sqrt(fabs(driftdistance/10.))/1000.;
90  double gaus_transx=gsl_ran_gaussian(r,difftrans*meandrift);
91  double gaus_transy=gsl_ran_gaussian(r,difftrans*meandrift);
92  double gaus_long=gsl_ran_gaussian(r,difflong*sqrt(fabs(driftdistance/10.))/1000.);
93  // for diffusion full tpc length, t=0 at fist GEMS, rest of transverse diff is done between GEMS in TPCPads
94  tdrift = t_e + (fabs(driftdistance)+gaus_long)/(vdrift); // [mus]
95  xdrift = x_e + gaus_transx; // [mm]
96  ydrift = y_e + gaus_transy;
97  zdrift = z_e + driftdistance;
98 }

335 double TrackDrifter::CalculateGasParameter(double a, double b ,double c, double d,
336  double e, double f, double g, double h,
337  double i, double j,double k)
338 {
339  double logE=log(E);
340  return a+b*logE+c*pow(logE,2)+d*pow(logE,3)
341  +e*pow(logE,4)+f*pow(logE,5)+g*pow(logE,6)
342  +h*pow(logE,7)+i*pow(logE,8)+j*pow(logE,9)
343  +k*pow(logE,10);
344 }
